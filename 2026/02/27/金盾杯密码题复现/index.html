方法：利用 dp 攻击

dp = d mod (p-1)

因为 d = e^-1 mod phi, 所以 dp = d - k*(p-1) 对于某个整数 k

我们可以尝试不同的 k 值来找到 p

def attack_rsa_dp(n, e, dp):
“””
使用已知的 dp 值攻击 RSA
dp = d mod (p-1)
“””
# 尝试不同的 k 值
for k in range(1, e): # k 应该在 1 到 e-1 之间
# dp = d - k*(p-1)
# d = dp + k*(p-1)

    # 因为 d*e ≡ 1 (mod phi)
    # 所以 (dp + k*(p-1))*e ≡ 1 (mod (p-1)*(q-1))
    
    # 这个同余式可以简化为寻找 p 的方程
    # 我们可以尝试另一种方法
    
    # 从 dp = d mod (p-1) 出发
    # 我们有 d = dp + k*(p-1) 对于某个 k
    # 代入 d*e ≡ 1 (mod phi)
    
    # 尝试计算可能的 p
    try:
        # 使用 Wiener's attack 的变种
        # 这里使用一个更简单的方法
        
        # 我们知道 dp = d mod (p-1)
        # 所以 d = dp + k*(p-1) 对于某个整数 k
        # 因为 d = e^-1 mod phi，所以 d*e = 1 + m*phi 对于某个整数 m
        
        # 代入得: (dp + k*(p-1))*e = 1 + m*(p-1)*(q-1)
        # 展开: dp*e + k*e*(p-1) = 1 + m*(p-1)*(q-1)
        # 重排: dp*e - 1 = (p-1)*(m*(q-1) - k*e)
        
        # 所以 (p-1) 是 (dp*e - 1) 的一个因子
        temp = dp * e - 1
        
        # 尝试找到 (p-1) 的候选值
        if temp % k == 0:
            possible_p_minus_1 = temp // k
            possible_p = possible_p_minus_1 + 1
            
            # 检查这个 p 是否能整除 n
            if n % possible_p == 0:
                p = possible_p
                q = n // p
                
                # 验证结果
                if p * q == n and p > 1 and q > 1:
                    print(f"找到因子!")
                    print(f"p = {p}")
                    print(f"q = {q}")
                    return p, q
    except:
        continue

return None, None

print(“正在尝试恢复 p 和 q…”)
p, q = attack_rsa_dp(n, e, dp)

if p and q:
print(f”\n成功分解 n!”)

print(“\n尝试另一种方法…”)

phi = (p - 1) * (q - 1)
d = pow(e, -1, phi)

m = pow(c, d, n)
